124     @Override
125     public void start(final boolean localOnly, final boolean anyAddress) {
126         final int port = getPort();
127         log.debug("Starting proxy on port: "+port);
128         final HttpServerPipelineFactory factory =
129             new StatsTrackingHttpServerPipelineFactory(authenticationManager,
130                 this.allChannels, this.chainProxyManager, 
131                 new StatsTrackingDefaultRelayPipelineFactoryFactory(chainProxyManager,
132                     this.responseFilters, this.requestFilter,
133                     this.allChannels, this.timer), 
134                 this.clientChannelFactory, this.timer, 
135                 this.handshakeHandlerFactory);
136         serverBootstrap.setPipelineFactory(factory);
137 
138         // Binding only to localhost can significantly improve the security of
139         // the proxy.
140         InetSocketAddress isa;
141         if (localOnly) {
142             isa = new InetSocketAddress("127.0.0.1", port);
143         }
144         else if (anyAddress) {
145             isa = new InetSocketAddress(port);
146         } else {
147             try {
148                 isa = new InetSocketAddress(NetworkUtils.getLocalHost(), port);
149             } catch (final UnknownHostException e) {
150                 log.error("Could not get local host?", e);
151                 isa = new InetSocketAddress(port);
152             }
153         }
154         final Channel channel = serverBootstrap.bind(isa);
155         allChannels.add(channel);
156 
157         Runtime.getRuntime().addShutdownHook(new Thread(new Runnable() {
158             @Override
159             public void run() {
160                 stop();
161             }
162         }));
163         log.debug("Started "+this);
164     }

180     /**
181      * This class is the pipeline factory for the initial requests themselves
182      * as opposed to the relayed requests. So this receives HTTP requests
183      * directly from peers, while StatsTrackingDefaultRelayPipelineFactoryFactory
184      * relays those requests to the destination site, of course relaying the
185      * corresponding responses back to peers.
186      */
187     private class StatsTrackingHttpServerPipelineFactory 
188         extends HttpServerPipelineFactory {
189         
190 
191         private StatsTrackingHttpServerPipelineFactory(
192             final ProxyAuthorizationManager authorizationManager, 
193             final ChannelGroup channelGroup, 
194             final ChainProxyManager chainProxyManager, 
195             final RelayPipelineFactoryFactory relayPipelineFactoryFactory,
196             final ClientSocketChannelFactory clientChannelFactory, 
197             final Timer timer,
198             final HandshakeHandlerFactory handshakeHandlerFactory) {
199             super(authorizationManager, channelGroup, chainProxyManager, 
200                     handshakeHandlerFactory, relayPipelineFactoryFactory, 
201                 timer, clientChannelFactory);
202         }
203 
204         @Override
205         public ChannelPipeline getPipeline() throws Exception {
206             final ChannelPipeline pipeline = super.getPipeline();
207             // Could be null for testing.
208             if (stats != null) {
209                 pipeline.addFirst("stats", new StatsTrackingHandler() {
210                     @Override
211                     public void addUpBytes(long bytes, final Channel ch) {
212                         stats.addUpBytesToPeers(bytes);
213                     }
214                     @Override
215                     public void addDownBytes(long bytes, final Channel ch) {
216                         stats.addDownBytesFromPeers(bytes);
217                         stats.addBytesProxied(bytes, ch);
218                     }
219                 });
220             }
221             
222             // This allows us to track global stats and also acts as our
223             // hook to add per-IP stats tracking.
224             // Could be null for testing.
225             if (serverTrafficHandler != null) {
226                 pipeline.addFirst("globalTrafficHandler", serverTrafficHandler);
227             }
228             return pipeline;
229         }
230     }

233      * This class tracks statistics for peers we are proxying for. This
234      * creates connections to remote servers and tracks the data going to and 
235      * from them as opposed to the data to and from the peer itself.
236      * 
237      * So we DON'T learn about any new Lantern peers here, but rather remote

239      */
240     private class StatsTrackingDefaultRelayPipelineFactoryFactory 
241         extends DefaultRelayPipelineFactoryFactory {
242         
243         private StatsTrackingDefaultRelayPipelineFactoryFactory(
244             final ChainProxyManager chainProxyManager, 
245             final HttpResponseFilters responseFilters, 
246             final HttpRequestFilter requestFilter, 
247             final ChannelGroup channelGroup,
248             final Timer timer) {
249             super(chainProxyManager, responseFilters, requestFilter, channelGroup, timer);
250         }
251         
252         @Override
253         public ChannelPipelineFactory getRelayPipelineFactory(
254             final HttpRequest httpRequest,final Channel browserToProxyChannel,
255             final RelayListener relayListener) {
256             final ChannelPipelineFactory innerFactory =
257                 super.getRelayPipelineFactory(httpRequest, 
258                     browserToProxyChannel, relayListener);
259 
260             return new ChannelPipelineFactory() {
261                 @Override
262                 public ChannelPipeline getPipeline() throws Exception {
263                     final ChannelPipeline pipeline = innerFactory.getPipeline();
264                     if (stats != null) {
265                         pipeline.addFirst("stats", new StatsTrackingHandler() {
266                             @Override
267                             public void addUpBytes(long bytes, final Channel ch) {
268                                 stats.addUpBytesForPeers(bytes);
269                             }
270                             @Override
271                             public void addDownBytes(long bytes, final Channel ch) {
272                                 stats.addDownBytesForPeers(bytes);
273                                 stats.addBytesProxied(bytes, ch);
274                             }
275                         });
276                     }
277                     return pipeline;
278                 }
279             };
280         }
