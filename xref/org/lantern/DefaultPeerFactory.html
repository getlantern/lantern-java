59      /**
60       * There are two ways we initially learn about new peers. The first is a
61       * Lantern peer directly on our roster, which will produce this event. The
62       * second is a kaleidoscope advertisement. Those Kaleidoscope advertisements
63       * can be from peers on our roster, but they can also be from peers we're
64       * not directly connected to. This method captures the first case where
65       * peers on our roster are running Lantern.
66       * 
67       * @param event
68       *            The update presence event.
69       */
70      @Subscribe
71      public void onUpdatePresenceEvent(final UpdatePresenceEvent event) {
72          log.debug("Processing presence event");
73          final Presence presence = event.getPresence();
74          final String from = presence.getFrom();
75          if (StringUtils.isBlank(from)) {
76              log.warn("Presence with blank from?");
77          } else {
78              addPeer(LanternUtils.newURI(from), Type.pc);
79          }
80      }

82      @Subscribe
83      public void onKscopeAd(final KscopeAdEvent event) {
84          final LanternKscopeAdvertisement ad = event.getAd();
85          // It is possible and even likely we already know about this peer
86          // through some other means, in which case we have to update the data
87          // about that peer as necessary.
88          log.debug("Adding peer through kscope ad...");
89          final String jid = ad.getJid();
90          final URI uri = LanternUtils.newURI(jid);
91          final Peer existing = this.model.getPeerCollector().getPeer(uri);
92          final LanternRosterEntry entry = this.roster.getRosterEntry(jid);
93          if (existing == null) {
94              // The following can be null.
95              final Peer peer = new Peer(uri, "",
96                      ad.hasMappedEndpoint(), 0, 0, Type.pc, ad.getAddress(),
97                      ad.getPort(), Mode.give, false, entry);
98              this.model.getPeerCollector().addPeer(uri, peer);
99              updateGeoData(peer, ad.getAddress());
100         } else {
101             existing.setIp(ad.getAddress());
102             existing.setPort(ad.getPort());
103             existing.setMode(Mode.give);
104             existing.setMapped(ad.hasMappedEndpoint());
105             if (existing.getRosterEntry() == null) {
106                 // Ours could be null too, but can't hurt to set.
107                 existing.setRosterEntry(entry);
108             }
109             existing.setVersion(ad.getLanternVersion());
110             updateGeoData(existing, ad.getAddress());
111         }
112     }

118     private void updateGeoData(final Peer peer, final String address) {
119         if (peer.hasGeoData()) {
120             log.debug("Peer already had geo data: {}", peer);
121             return;
122         }
123 
124         final GeoData geo = geoIpLookupService.getGeoData(address);
125         peer.setCountry(geo.getCountrycode());
126         peer.setLat(geo.getLatitude());
127         peer.setLon(geo.getLongitude());
128     }

130     private void updatePeer(final URI fullJid, final InetSocketAddress isa,
131             final Type type) {
132         final Peer peer = this.model.getPeerCollector().getPeer(fullJid);
133         if (peer == null) {

140     private void updatePeer(final Peer peer, final InetSocketAddress isa,
141             final Type type) {
142         final String address = isa.getAddress().getHostAddress();
143         if (StringUtils.isBlank(peer.getIp())) {
144             peer.setIp(address);
145         }
146         if (peer.getPort() == 0) {
147             peer.setPort(isa.getPort());
148         }
149         if (peer.getRosterEntry() == null) {
150             log.debug("Setting roster entry");
151             final URI uri = LanternUtils.newURI(peer.getPeerid());
152             peer.setRosterEntry(rosterEntry(uri));
153         }
154         peer.setType(type.toString());
155         updateGeoData(peer, isa.getAddress());
156         // Note we don't sync peers with the frontend here because the timer
157         // will do it for us
158     }

166     @Override
167     public Peer addPeer(final URI fullJid, final Type type) {
168 
169         // This is a peer we know very little about at this point, as we
170         // haven't made any network connections with them.
171         final LanternRosterEntry entry = rosterEntry(fullJid);
172         log.debug("Got roster entry: {} for '{}'", entry, fullJid);
173         if (entry == null) {
174             // This will happen for cloud "peers" and kscope peers but otherwise
175             // all peers should be on your roster.
176             log.debug("Could not find match for type '{}':\n{}", type, 
177                     ThreadUtils.dumpStack());
178             log.debug("Roster is: {}", this.roster.getEntries());
179         }
180 
181         final Peer existing = this.model.getPeerCollector().getPeer(fullJid);
182 
183         if (existing != null) {
184             log.debug("Peer already exists...");
185             return existing;
186         } else {
187             log.debug("Adding peer {}", fullJid);
188             final Peer peer = new Peer(fullJid, "", false, 0L, 0L, type,
189                     "", 0, Mode.unknown, false, entry);
190             this.model.getPeerCollector().addPeer(fullJid, peer);
191             return peer;
192         }
193     }

204     private final Map<X509Certificate, Peer> certsToPeers =
205             new ConcurrentHashMap<X509Certificate, Peer>();
206 
207     @Subscribe
208     public void onCert(final PeerCertEvent event) {
209         final Peer peer = this.model.getPeerCollector().getPeer(event.getJid());
210         if (peer == null) {
211             log.error("Got a cert for peer we don't know about? " +
212                     "{} not found in {}", event.getJid(),
213                     this.model.getPeerCollector().getPeers().keySet());
214         } else {
215             byte[] certificateBytes = Base64
216                     .decodeBase64(event.getBase64Cert());
217             try {
218                 X509Certificate certificate = X509Certificate
219                         .getInstance(certificateBytes);
220                 certsToPeers.put(certificate, peer);
221             } catch (CertificateException ce) {
222                 log.error("Unable to decode X509 certificate", ce);
223             }
224         }
225     }

227     @Override
228     public Peer peerForSession(SSLSession sslSession) {
229         try {
230             X509Certificate[] certificateChain = sslSession
231                     .getPeerCertificateChain();
232             if (certificateChain.length == 0) {
233                 log.error("No certificates in chain");
234                 return null;
235             }
236             X509Certificate cert = certificateChain[0];
237             return certsToPeers.get(cert);
238         } catch (SSLPeerUnverifiedException spue) {
239             log.error("Peer not verified");
240             return null;
241         }
242     }

244     public void peerSentRequest(InetSocketAddress peerAddress,
245             SSLSession sslSession) {
246         Peer peer = peerForSession(sslSession);
247         if (peer != null) {
248             log.debug("Found peer by certificate!!!");
249             peer.setMode(Mode.get);
250             updatePeer(peer, peerAddress, Type.pc);
251         } else {
252             log.error("No peer found for ssl session: {}", sslSession);
253         }
