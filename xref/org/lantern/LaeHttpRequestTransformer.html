11  /**
12   * Modified HTTP requests to work with Google App Engine as a proxy.
13   */
14  public class LaeHttpRequestTransformer implements HttpRequestTransformer {
15  
16      private final Logger log = LoggerFactory.getLogger(getClass());
17      
18      @Override
19      public void transform(final HttpRequest request,
20          final InetSocketAddress proxyAddress) {
21          final String uri = request.getUri();
22          
23          // TODO: This call can do a reverse DNS lookup here. Do we really want
24          // this? Do we already have the IP in this case? 
25          // Use getAddress().getHostAddress() instead?
26          final String host = proxyAddress.getHostName();
27          final String proxyBaseUri = "https://" + host;
28          if (!uri.startsWith(proxyBaseUri)) {
29              request.setHeader("Host", host);
30              final String scheme = uri.substring(0, uri.indexOf(':'));
31              final String rest = uri.substring(scheme.length() + 3);
32              final String proxyUri = proxyBaseUri + "/" + scheme + "/" + rest;
33              log.debug("proxyUri: " + proxyUri);
34              request.setUri(proxyUri);
35          } else {
36              log.info("NOT MODIFYING URI -- ALREADY HAS HOST");
37          }
38          
39          final String range = request.getHeader(HttpHeaders.Names.RANGE);
40          if (StringUtils.isNotBlank(range)) {
41              log.info("Request already has range!");
42              return;
43          }
44          
45          request.setHeader(LanternConstants.LANTERN_VERSION_HTTP_HEADER_NAME, 
46                  LanternClientConstants.LANTERN_VERSION_HTTP_HEADER_VALUE);
47          request.setHeader(HttpHeaders.Names.RANGE, 
48              "bytes=0-"+LanternConstants.CHUNK_SIZE);
49      }
