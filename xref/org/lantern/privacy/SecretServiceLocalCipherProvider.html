65      public static boolean secretServiceAvailable() {
66          DBusConnection conn = null;
67          try {
68              LOG.debug("Checking for Secret Service API availability...");
69              conn = DBusConnection.getConnection(DBusConnection.SESSION);
70              final Service secretService = conn.getRemoteObject(
71                  BUS_NAME, SERVICE_PATH, Service.class);
72              final Pair<Variant, Path> result = 
73                  secretService.OpenSession(ALGORITHM_PLAIN, new Variant(""));
74              if (NO_OBJECT.equals(result.b.getPath())) {
75                  LOG.info("Failed to negotiate plain session with Secret Service API.");
76                  return false;
77              }
78              else {
79                  final Session session = conn.getRemoteObject(
80                      BUS_NAME, result.b.getPath(), Session.class);
81                  session.Close();
82              }
83              
84              LOG.debug("Found Secret Service API.");
85              return true;
86          } catch (final Exception e) {
87              LOG.debug("Could not connect to Secret Service API at:"+SERVICE_PATH, e);
88              return false;
89          } catch (final UnsatisfiedLinkError error) {
90              if (LanternUtils.isDevMode()) {
91                  LOG.error("Could not load so file? Testing?", error);
92                  return false;
93              } else {
94                  LOG.error("CAN'T HAVE UNENCRYPTED FILES IN PRODUCTION!! " +
95                      "\njava.library.path is: "+
96                      System.getProperty("java.library.path")+
97                      "\nAll props: "+System.getProperties(), error);
98                  System.exit(1);
99                  return false;
100             }
101         } finally {
102             closeConnection(conn);
103         }

116     @Override
117     byte[] loadKeyData() throws IOException, GeneralSecurityException {
118         DBusConnection conn = null;
119         byte [] encodedKey = null;
120         try {
121             LOG.debug("Loading key data from Secret Service API...");
122 
123             conn = DBusConnection.getConnection(DBusConnection.SESSION);
124             final Service secretService = conn.getRemoteObject(
125                 BUS_NAME, SERVICE_PATH, Service.class);
126 
127             // negotiate a "plain" session (no encryption)
128             final Pair<Variant, Path> result = secretService.OpenSession(ALGORITHM_PLAIN, new Variant(""));
129             if (NO_OBJECT.equals(result.b.getPath())) {
130                 throw new IOException("Unable to negotiate DBus session!");
131             }
132             final Path sessionPath = result.b;
133             final Session session = conn.getRemoteObject(BUS_NAME, sessionPath.getPath(), Session.class);
134 
135             LOG.debug("Requesting Item....");
136             final Map<String, String> secretAttrs = new HashMap<String, String>();
137             secretAttrs.put(SECRET_ATTR_NAME,
138                             SECRET_ATTR_VALUE);
139             final Pair<List<Path>, List<Path>> items = secretService.SearchItems(secretAttrs);
140             LOG.debug("Got {} unlocked / {} locked secret items.", items.a.size(), items.b.size());
141 
142             Item secretItem = null;
143             if (items.a.isEmpty()) {
144                 if (items.b.isEmpty()) {
145                     LOG.debug("Secret item is missing!");
146                     throw new GeneralSecurityException("Unable to locate secret in keychain!");
147                 }
148                 else {
149                     LOG.debug("Secret is locked! Attempting to unlock...");
150                     final Path secretItemPath = items.b.get(0);
151                     if (unlockPath(secretItemPath, secretService, conn)) {
152                         LOG.debug("Assuming all is well using path {}...", secretItemPath.getPath());
153                         secretItem = conn.getRemoteObject(BUS_NAME, secretItemPath.getPath(), Item.class);
154                     }
155                     else {
156                         LOG.error("Failed to unlock lantern keychain item.");
157                         throw new GeneralSecurityException("Failed to unlock lantern keychain item.");
158                     }
159                 }
160             }
161             else {
162                 LOG.debug("Secret is unlocked...");
163                 final Path itemPath = items.a.get(0);
164                 secretItem = conn.getRemoteObject(BUS_NAME, itemPath.getPath(), Item.class);
165             }
166             final Secret secret = secretItem.GetSecret(sessionPath);
167             encodedKey = new byte[secret.value.size()];
168             int i = 0;
169             for (final Byte b : secret.value) {
170                 encodedKey[i++] = b;
171             }
172             session.Close();
173 
174             final Base64 base64 = new Base64();
175             return base64.decode(encodedKey);
176         } catch (DBusException e) {
177             throw new IOException(e);
178         } catch (InterruptedException e) {
179             throw new IOException(e);
180         }finally {
181             if (encodedKey != null) {
182                 Arrays.fill(encodedKey, (byte) 0);
183             }
184             closeConnection(conn);
185         }
186     }

188     @Override
189     void storeKeyData(final byte[] key) throws IOException, GeneralSecurityException {
190         byte [] encodedKey = null;
191         List<Byte> secretValue = null;
192 
193         DBusConnection conn = null;
194         try {
195             LOG.debug("Storing key data via Secret Service API...");
196             conn = DBusConnection.getConnection(DBusConnection.SESSION);
197             final Service secretService = conn.getRemoteObject(
198                 BUS_NAME, SERVICE_PATH, Service.class);
199 
200             final Collection collection = conn.getRemoteObject(
201                 BUS_NAME, COLLECTION_PATH, Collection.class);
202 
203             // Collection properties do not seem to be implemented...
204             // So instead of checking, just request an unlock in all cases
205             if (!unlockPath(new Path(COLLECTION_PATH), secretService, conn)) {
206                 throw new GeneralSecurityException("Unable to unlock secret collection.");
207             }
208 
209             // negotiate a "plain" session (no encryption)
210             final Pair<Variant, Path> result = secretService.OpenSession(ALGORITHM_PLAIN, new Variant(""));
211             if (NO_OBJECT.equals(result.b.getPath())) {
212                 throw new IOException("Unable to negotiate DBus session!");
213             }
214             final Path sessionPath = result.b;
215             final Session session = conn.getRemoteObject(
216                 BUS_NAME, sessionPath.getPath(), Session.class);
217 
218             // construct a "Secret"
219             final Base64 base64 = new Base64();
220             encodedKey = base64.encode(key);
221             secretValue = new ArrayList<Byte>(encodedKey.length);
222             for (byte b : encodedKey) {
223                 secretValue.add(b);
224             }
225             final Secret secret = new Secret(sessionPath, new ArrayList<Byte>(), secretValue, SECRET_CONTENT_TYPE);
226             final Map<String, Variant> secretProps = new HashMap<String, Variant>();
227             secretProps.put(SECRET_LABEL_PROPERTY,
228                             new Variant(LANTERN_KEY_LABEL));
229 
230             // these are the "attributes" of the item, not to be confused
231             // with properties...
232             final Map<String, String> secretAttrs = new HashMap<String, String>();
233             secretAttrs.put(SECRET_ATTR_NAME, SECRET_ATTR_VALUE);
234             secretProps.put(SECRET_ATTRIBUTES_PROPERTY, new Variant(secretAttrs, "a{ss}"));
235 
236             // add the item to the collection...
237             LOG.debug("Requesting CreateItem....");
238             final Pair<Path, Path> createResult = collection.CreateItem(
239                 secretProps, secret, true);
240 
241             final Path promptPath = createResult.b;
242 
243             // user may need to be prompted... for what is unclear.
244             if (!NO_OBJECT.equals(promptPath.getPath())) {
245                 Prompt.Completed sig = prompt(promptPath, conn);
246                 // there does not seem to be a specification of what this means in this case... ?
247                 // possibly the path to the item or / on failure?
248                 LOG.debug("Prompt completed with dismissed={} result={}", sig.dismissed, sig.result);
249                 final List<Path> createdItemPaths = (List<Path>) sig.result.getValue();
250                 Path createdItemPath = null;
251                 if (createdItemPaths.size() > 0 ) {
252                     createdItemPath = createdItemPaths.get(0);
253                 }
254                 if (sig.dismissed == true || createdItemPath == null || NO_OBJECT.equals(createdItemPath.getPath())) {
255                     LOG.error("Failed to create lantern keychain item.");
256                     throw new GeneralSecurityException("Failed to create lantern keychain item.");
257                 }
258             }
259             else {
260                 LOG.debug("No prompt requested.");
261             }
262 
263             session.Close();
264             LOG.debug("Finished storing key in Secret Service API...");
265         } catch (DBusException e) {
266             throw new IOException(e);
267         } catch (InterruptedException e) {
268             throw new IOException(e);
269         } finally {
270             if (encodedKey != null) {
271                 Arrays.fill(encodedKey, (byte) 0);
272             }
273             closeConnection(conn);
274         }

277     private boolean unlockPath(final Path path, final Service secretService, final DBusConnection conn)
278         throws DBusException, InterruptedException {
279         LOG.debug("Requesting unlock of path {}", path.getPath());
280         final List<Path> pathsToUnlock = new ArrayList<Path>(1);
281         pathsToUnlock.add(path);
282         final Pair<List<Path>, Path> unlockResult = secretService.Unlock(pathsToUnlock);
283         if (unlockResult.a.size() > 0) {
284             LOG.debug("Path unlocked without prompt...");
285             return true;
286         }
287         // if there is a prompt
288         else if (!NO_OBJECT.equals(unlockResult.b.getPath())) {
289             LOG.debug("Prompting user to unlock....");
290             Prompt.Completed sig = prompt(unlockResult.b, conn);
291             LOG.debug("Prompt completed with dismissed={} result={}", sig.dismissed, sig.result);
292             final List<Path> unlockedPaths = (List<Path>) sig.result.getValue();
293             Path unlockedPath = null;
294             if (unlockedPaths.size() > 0) {
295                 unlockedPath = unlockedPaths.get(0);
296             }
297             if (sig.dismissed == true || unlockedPath == null || NO_OBJECT.equals(unlockedPath.getPath())) {
298                 LOG.error("Failed to unlock path {}", path.getPath());
299                 return false;
300             }
301             else {
302                 return true;
303             }
304         }
305         else {
306             LOG.debug("Confused, no prompt and no unlocked objects, failing.");
307             return false;
308         }

311     private Prompt.Completed prompt(final Path promptPath, final DBusConnection conn)
312         throws InterruptedException, DBusException {
313         LOG.debug("Prompt required...");
314         final Prompt prompt = conn.getRemoteObject(
315             BUS_NAME, promptPath.getPath(), Prompt.class);
316 
317         final PromptHandler handler = new PromptHandler();
318         conn.addSigHandler(Prompt.Completed.class, prompt, handler);
319         LOG.debug("added signal handler, calling prompt");
320         try {
321             prompt.Prompt("0"); // XXX window-id ?
322             LOG.debug("Waiting for prompt to complete...");
323             return handler.await();
324         } finally {
325             conn.removeSigHandler(Prompt.Completed.class, prompt, handler);
326         }
327     }

329     /**
330      * This is a DBusSigHandler that waits for the
331      * response to a user prompt by sitting on a
332      * Condition variable.
333      */
334     private class PromptHandler implements DBusSigHandler {
335 
336         private final Lock lock;
337         private final Condition gotResult;
338         private DBusSignal sig = null;
339 
340         public PromptHandler() {
341             lock = new ReentrantLock();
342             gotResult = lock.newCondition();
343         }
344 
345         @Override
346         public void handle(final DBusSignal sig) {
347             this.sig = sig;
348             lock.lock();
349             try {
350                 gotResult.signal();
351             } finally {
352                 lock.unlock();
353             }
354         }
355 
356         public Prompt.Completed await() throws InterruptedException {
357             lock.lock();
358             try {
359                 while (sig == null) {
360                     gotResult.await();
361                 }
362                 return (Prompt.Completed) sig;
363             } finally {
364                 lock.unlock();
365             }
366         }
