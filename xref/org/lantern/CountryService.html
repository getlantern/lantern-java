275     private void initCountries() {
276         for (int i = 0; i < countryTable.length; i += 2) {
277             String countryCode = countryTable[i];
278             String countryName = countryTable[i + 1];
279             boolean isCensored = censored.isCountryCodeCensored(countryCode);
280             allCountries.put(countryCode, new Country(countryCode, countryName,
281                     isCensored));
282         }
283     }
284 
285     public Map<String, Country> allCountries() {

287     }
