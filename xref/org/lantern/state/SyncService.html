55      @Inject
56      public SyncService(final SyncStrategy strategy,
57          final Model model, final Timer timer,
58          final Transfers transfers) {
59          this.strategy = strategy;
60          this.model = model;
61          this.timer = timer;
62          this.transfers = transfers;
63          // Make sure the config class is added as a listener before this class.
64          Events.register(this);
65      }

68      @Override
69      public void start() {
70          if (timer != null) {
71              timer.schedule(new TimerTask() {
72                  @Override
73                  public void run() {
74                      //sync();
75                      delegateSync(SyncType.ADD, SyncPath.PEERS,
76                          model.getPeers());
77                      if (transfers != null) {
78                          delegateSync(SyncType.ADD, SyncPath.TRANSFERS,
79                                  transfers);
80                      }
81                  }
82              }, 3000, LanternClientConstants.SYNC_INTERVAL_SECONDS * 1000);
83          }
84      }
85  
86      @Override
87      public void stop() {
88          if (this.timer != null) {
89              this.timer.cancel();
90          }
91      }

100     public void metaConnect(final ServerSession remote, final Message connect) {
101         // Make sure we give clients the most recent data whenever they connect.
102         log.debug("Got connection from client, calling sync");
103 
104         final Thread t = new Thread(new Runnable() {
105             @Override
106             public void run() {

108                 delegateSync(SyncPath.ALL, model);

111         }, "CometD-Sync-OnConnect-Thread");
112         t.setDaemon(true);

116     @Subscribe
117     public void onSync(final SyncEvent syncEvent) {
118         log.debug("Got sync event");
119         // We want to force a sync here regardless of whether or not we've
120         // recently synced.
121         //sync(true, syncEvent.getChannel());
122         delegateSync(syncEvent.getOp(), syncEvent.getPath(), syncEvent.getValue());
123     }

125     @Subscribe
126     public void closedBeta(final ClosedBetaEvent betaEvent) {
127         final boolean alreadyInvited = this.model.getConnectivity().isInvited();
128 
129         final boolean invited = betaEvent.isInClosedBeta();
130         if (alreadyInvited == invited) {
131             log.debug("No change in invited state");
132             return;
133         }
134         // Note this is the only place setInvited should be called. We do all
135         // checks here to know whether or not to sync with the frontend and
136         // because of the use of ClosedBetaEvent for thread syncing in
137         // the xmpp handler.
138         this.model.getConnectivity().setInvited(invited);
139 
140         delegateSync(SyncPath.INVITED, invited);
141     }
