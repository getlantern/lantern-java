28      @Subscribe
29      public void onP2PConnectionEvent(P2PConnectionEvent e) {
30          if (e.isConnected()) {
31              // resend invites
32              Friends friends = model.getFriends();
33              ArrayList<String> pendingInvites = new ArrayList<String>(
34                      model.getPendingInvites());
35              for (String email : pendingInvites) {
36                  log.info("Resending pending invite to {}", email);
37                  Friend friend = friends.get(email);
38                  xmppHandler.sendInvite(friend, true);
39              }
40          }
41  
42      }
43  
44      public void invite(Friend friend) {
45          String email = friend.getEmail();
46          try {
47              if (xmppHandler.sendInvite(friend, false)) {
48                  // we need to mark this email as pending, in case
49                  // our invite gets lost.
50                  model.addPendingInvite(email);
51              }
52          } catch (Exception e) {
53              log.debug("failed to send invite: ", e);
54              model.addPendingInvite(email);
55          }
56          Events.sync(SyncPath.FRIENDS, model.getFriends().getFriends());
57  
58      }
