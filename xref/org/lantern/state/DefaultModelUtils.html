43      /**
44       * This is used for when the user disconnects and reconnects for any reason.
45       * We store the users we know to have been in the closed beta so we don't
46       * need to wait for a response from the lantern XMPP bot if we already
47       * know they're invited.
48       * 
49       * Note the user will typically only be connecting with one IP addres, but
50       * this setup ensures that any architecture changes the may change that 
51       * won't affect invite lookups.
52       * 
53       * @param email The email to check.
54       * @return <code>true</code> if we already know the specified user to be in
55       * the closed beta. This will return <code>false</code> if we just don't
56       * know -- they could be in or the could not be, but we haven't verified
57       * they are in.
58       */
59      @Override

61          final Set<String> in = this.model.getSettings().getInClosedBeta();

112     @Override
113     public void loadOAuth2ClientSecretsFile(final String filename) {
114         if (StringUtils.isBlank(filename)) {
115             LOG.error("No filename specified");
116             throw new NullPointerException("No filename specified!");
117         }
118         final File file = new File(filename);
119         if (!(file.exists() && file.canRead())) {
120             LOG.error("Unable to read user credentials from {}", filename);
121             throw new IllegalArgumentException("File does not exist! "+filename);
122         }
123         LOG.debug("Reading client secrets from file \"{}\"", filename);
124         try {
125             final String json = FileUtils.readFileToString(file, "US-ASCII");
126             JSONObject obj = (JSONObject)JSONValue.parse(json);
127             final JSONObject ins;
128             final JSONObject temp = (JSONObject)obj.get("installed");
129             if (temp == null) {
130                 ins = (JSONObject)obj.get("web");
131             } else {
132                 ins = temp;
133             }
134             //JSONObject ins = (JSONObject)obj.get("installed");
135             final String clientID = (String)ins.get("client_id");
136             final String clientSecret = (String)ins.get("client_secret");
137             if (StringUtils.isBlank(clientID) || 
138                 StringUtils.isBlank(clientSecret)) {
139                 LOG.error("Failed to parse client secrets file \"{}\"", file);
140                 throw new Error("Failed to parse client secrets file: "+ file);
141             } else {
142                 this.model.getSettings().setClientID(clientID);
143                 this.model.getSettings().setClientSecret(clientSecret);
144             }
145         } catch (final IOException e) {
146             LOG.error("Failed to read file \"{}\"", filename);
147             throw new Error("Could not load oauth file"+file, e);
148         }
149     }

151     @Override
152     public void loadOAuth2UserCredentialsFile(final String filename) {
153         if (StringUtils.isBlank(filename)) {
154             LOG.error("No filename specified");
155             throw new NullPointerException("No filename specified!");
156         }
157         final File file = new File(filename);
158         if (!(file.exists() && file.canRead())) {
159             LOG.error("Unable to read user credentials from {}", filename);
160             throw new IllegalArgumentException("File does not exist! "+filename);
161         }
162         LOG.info("Reading user credentials from file \"{}\"", filename);
163         try {
164             final String json = FileUtils.readFileToString(file, "US-ASCII");
165             final JSONObject obj = (JSONObject)JSONValue.parse(json);
166             final String username = (String)obj.get("username");
167             final String accessToken = (String)obj.get("access_token");
168             final String refreshToken = (String)obj.get("refresh_token");
169             // Access token is not strictly necessary, so we allow it to be
170             // null.
171             if (StringUtils.isBlank(username) || 
172                 StringUtils.isBlank(refreshToken)) {
173                 LOG.error("Failed to parse user credentials file \"{}\"", filename);
174                 throw new Error("Could not load username or refresh_token");
175             } else {
176                 //this.model.getSettings().setEmail(username);
177                 //this.model.getSettings().setCommandLineEmail(username);
178                 this.model.getSettings().setAccessToken(accessToken);
179                 this.model.getSettings().setRefreshToken(refreshToken);
180                 this.model.getSettings().setUseGoogleOAuth2(true);
181             }
182         } catch (final IOException e) {
183             LOG.error("Failed to read file \"{}\"", filename);
184             throw new Error("Could not load oauth credentials", e);
185         }
186     }

188     @Override
189     public LanternGoogleOAuth2Credentials newGoogleOauthCreds(final String resource) {
190         final Settings set = this.model.getSettings();
191         if (LanternUtils.isDevMode()) {
192             final File oauth = LanternClientConstants.TEST_PROPS;
193             if (!oauth.isFile()) {
194                 final Properties props = new Properties();
195                 props.put("refresh_token", set.getRefreshToken());
196                 props.put("access_token", set.getAccessToken());
197                 OutputStream os = null;
198                 try {
199                     os = new FileOutputStream(oauth);
200                     props.store(os, "Automatically stored test oauth tokens");
201                 } catch (final IOException e) {
202                 } finally {
203                     IOUtils.closeQuietly(os);
204                 }
205             } else {
206                 LOG.info("Not overwriting existing oauth file.");
207             }
208         }
209         return new LanternGoogleOAuth2Credentials("anon@getlantern.org",
210             set.getClientID(), set.getClientSecret(), 
211             set.getAccessToken(), set.getRefreshToken(), 
212             resource);
213     }

234     @Override
235     public Friend makeFriend(String email) {
236         Friends friends = model.getFriends();
237         Friend friend = friends.get(email);
238         if (friend == null) {
239             friend = new Friend(email);
240             Roster roster = model.getRoster();
241             final RosterEntry entry = roster.getEntry(email);
242             if (entry != null) {
243                 friend.setName(entry.getName());
244             }
245             friends.add(friend);
246         }
247         return friend;
