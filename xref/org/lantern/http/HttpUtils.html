52      /**
53       * Converts the request arguments to a map of parameter keys to single
54       * values, ignoring multiple values.
55       * 
56       * @param req The request.
57       * @return The mapped argument names and values.
58       */
59      public static Map<String, String> toParamMap(final ServletRequest req) {
60          final Map<String, String> map = new TreeMap<String, String>(
61                  String.CASE_INSENSITIVE_ORDER);
62          final Map<String, String[]> paramMap = req.getParameterMap();
63          final Set<Entry<String, String[]>> entries = paramMap.entrySet();
64          for (final Entry<String, String[]> entry : entries) {
65              final String[] values = entry.getValue();
66              map.put(entry.getKey(), values[0]);
67          }
68          return map;
69      }

81      /**
82       * Gets request headers as a string.
83       * 
84       * @param request The request.
85       * @return The request headers as a string.
86       */
87      public static String getRequestHeaders(final HttpServletRequest request) {
88          final Enumeration<String> headers = request.getHeaderNames();
89          final StringBuilder sb = new StringBuilder();
90          sb.append("\n");
91          while (headers.hasMoreElements()) {
92              final String headerName = headers.nextElement();
93              sb.append(headerName);
94              sb.append(": ");
95              sb.append(request.getHeader(headerName));
96              sb.append("\n");
97          }
98          return sb.toString();
99      }

102     /**
103      * Creates a map of header names to values from an HTTP request.
104      * 
105      * @param request The HTTP request.
106      * @return The map of header names to values.
107      */
108     public static Map<String, String> toHeaderMap(
109             final HttpServletRequest request) {
110         final Map<String, String> map = new TreeMap<String, String>(
111                 String.CASE_INSENSITIVE_ORDER);
112         final Enumeration<String> headers = request.getHeaderNames();
113         while (headers.hasMoreElements()) {
114             final String name = headers.nextElement();
115             final String value = request.getHeader(name);
116             map.put(name, value);
117         }
118         return map;
119     }
