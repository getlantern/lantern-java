107     private void createKeyStore() {
108         // Whenever the key store changes, we need to set the ref to null
109         // so we don't pass stale versions of the key manager factory 
110         // to callers.
111         
112         this.keyManagerFactoryRef.set(null);
113         final String dummyId = String.valueOf(RandomUtils.nextInt());
114         // Generate the keystore using a dummy ID.
115         log.debug("Dummy ID is: {}", dummyId);
116         log.debug("Creating keystore...");
117         
118         String result = LanternUtils.runKeytool("-genkey", 
119             "-alias", dummyId, 
120             "-keysize", KEYSIZE, 
121             "-validity", "365", 
122             "-keyalg", ALG, 
123             "-dname", "CN="+dummyId, 
124             "-keypass", PASS, 
125             "-storepass", PASS, 
126             "-keystore", KEYSTORE_FILE.getAbsolutePath());
127         log.debug("Got response: {}", result);
128         
129         log.debug("Deleting dummy alias...");
130         result = LanternUtils.runKeytool("-delete", "-alias", dummyId,
131             "-keypass", PASS, "-storepass", PASS,
132             "-keystore", KEYSTORE_FILE.getAbsolutePath());
133         log.debug("Got response: {}", result);
134         
135         log.debug("Done creating keystore...");
136     }

138     private void generateLocalCert(final String jid) {
139         // Whenever the key store changes, we need to set the ref to null
140         // so we don't pass stale versions of the key manager factory 
141         // to callers.
142         this.keyManagerFactoryRef.set(null);
143         final String genKeyResult = LanternUtils.runKeytool("-genkey", 
144             "-alias", jid, 
145             "-keysize", KEYSIZE, 
146             "-validity", "365", 
147             "-keyalg", ALG, 
148             "-dname", "CN="+jid,
149             "-keypass", PASS,
150             "-storepass", PASS,
151             "-keystore", KEYSTORE_FILE.getAbsolutePath());
152         
153         log.debug("Result of keytool -genkey call: {}", genKeyResult);
154         
155         // Now grab our newly-generated cert. All of our trusted peers will
156         // use this to connect.
157         // See: 
158         // http://docs.oracle.com/javase/6/docs/technotes/tools/solaris/keytool.html
159         // for a discussion of export versus exportcert (basically we use
160         // export for backwards compatibility).
161         final String exportCertResult = LanternUtils.runKeytool(
162             "-export", 
163             "-alias", jid, 
164             "-keystore", KEYSTORE_FILE.getAbsolutePath(), 
165             "-storepass", PASS, 
166             "-file", CERT_FILE.getAbsolutePath());
167         
168         log.debug("Result of keytool -exportcert call: {}", exportCertResult);
169         LanternUtils.waitForFile(CERT_FILE);
170         
171         try {
172             final InputStream is = new FileInputStream(CERT_FILE);
173             localCert = Base64.encodeBase64String(IOUtils.toByteArray(is));
174         } catch (final FileNotFoundException e) {
175             log.error("Cert file not found at "+CERT_FILE, e);
176             throw new Error("Cert file not found", e);
177         } catch (final IOException e) {
178             log.error("Could not base 64 encode cert?", e);
179             throw new Error("Could not base 64 encode cert?", e);
180         }
181     }

185         // The keystore file itself is created lazily, so make sure we have it
186         // before proceeding here.

225     public KeyManagerFactory getKeyManagerFactory() {
226         if (this.keyManagerFactoryRef.get() != null) {
227             return this.keyManagerFactoryRef.get();
228         }
229         String algorithm =
230             Security.getProperty("ssl.KeyManagerFactory.algorithm");
231         if (algorithm == null) {
232             algorithm = "SunX509";
233         }
234         try {
235             final KeyStore ks = KeyStore.getInstance("JKS");
236             ks.load(keyStoreAsInputStream(), getKeyStorePassword());
237 
238             // Set up key manager factory to use our key store
239             final KeyManagerFactory kmf = KeyManagerFactory.getInstance(algorithm);
240             kmf.init(ks, getKeyStorePassword());
241 
242             this.keyManagerFactoryRef.set(kmf);
243             return kmf;
244         } catch (final KeyStoreException e) {
245             throw new Error("Key manager issue", e);
246         } catch (final UnrecoverableKeyException e) {
247             throw new Error("Key manager issue", e);
248         } catch (final NoSuchAlgorithmException e) {
249             throw new Error("Key manager issue", e);
250         } catch (final CertificateException e) {
251             throw new Error("Key manager issue", e);
252         } catch (final IOException e) {
253             throw new Error("Key manager issue", e);
254         }
