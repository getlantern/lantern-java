11  /**
12   * HTTP response encoder for Lantern that sends responses back to the browser
13   * and keeps statistics on bytes proxied.
14   */
15  public class LanternHttpResponseEncoder extends ProxyHttpResponseEncoder {
16      
17      private final Logger log = LoggerFactory.getLogger(getClass());
18      private final ClientStats stats;
19  
20  
21      public LanternHttpResponseEncoder(final ClientStats stats) {
22          super(true);
23          this.stats = stats; 
24      }
25  
26      @Override
27      protected Object encode(final ChannelHandlerContext ctx, 
28          final Channel channel, final Object msg) throws Exception {
29          log.debug("Encoding response...");
30          final ChannelBuffer cb = 
31              (ChannelBuffer) super.encode(ctx, channel, msg);
32          if (cb == null) {
33              return null;
34          }
35          final int bytes = cb.readableBytes();
36          if (msg instanceof ProxyHttpResponse) {
37              // This is called when we just pass unproxied requests over to
38              // LittleProxy.
39              this.stats.addDirectBytes(bytes);
40          } else {
41              // If it's *not* a ProxyHttpResponse, that means it's something
42              // we didn't simply pass to LittleProxy and instead proxied
43              // ourselves. This could be a straight ChannelBuffer from a p2p
44              // socket, relayed non-HTTP CONNECT data from one of our proxies.
45              // HTTP CONNECT data has to be accounted for differently, as it
46              // bypassed any encoder.
47              
48              // global bytes proxied statistic
49              log.debug("Adding bytes proxied!!!");
50              this.stats.addBytesProxied(bytes, channel);
51          }
52          return cb;
53      }
