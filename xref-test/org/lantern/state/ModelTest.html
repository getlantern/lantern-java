24      @Test
25      public void testStartAtLogin() throws Exception {
26          if (!LanternClientConstants.LAUNCHD_PLIST.isFile()) {
27              log.info("No plist file - not installed or on different OS?");
28              return;
29          }
30          final Model model = TestUtils.getModel();
31          final Proxifier proxifier = TestUtils.getProxifier();
32          final ModelUtils modelUtils = TestUtils.getModelUtils();
33          final DefaultXmppHandler xmppHandler = TestUtils.getXmppHandler();
34          final ModelIo modelIo = TestUtils.getModelIo();
35          final Settings settings = model.getSettings();
36          final Roster roster = TestUtils.getRoster();
37          final File temp1 = plist();
38          final File temp2 = autostart();
39          //final File settingsFile = settingsFile();
40          FileUtils.copyFile(new File("install/osx/org.lantern.plist"), temp1);
41          FileUtils.copyFile(new File("install/linux/lantern-autostart.desktop"), temp2);
42          final String cur1 = FileUtils.readFileToString(temp1, "UTF-8");
43          final String cur2 = FileUtils.readFileToString(temp2, "UTF-8");
44          assertTrue(cur1.contains("<true/>") || cur1.contains("<false/>"));
45          assertTrue(cur2.contains("X-GNOME-Autostart-enabled=true") ||
46                  cur2.contains("X-GNOME-Autostart-enabled=false"));
47          //final SettingsIo ss = new SettingsIo(settingsFile, encryptedFileService);
48          final DefaultModelService implementor =
49              new DefaultModelService(temp1, temp2, model, proxifier);
50          if (cur1.contains("<true/>")) {
51              assertFalse(cur1.contains("<false/>"));
52              //Configurator.setStartAtLogin(temp, false);
53              settings.setRunAtSystemStart(false);
54              implementor.setStartAtLoginOsx(false);
55              modelIo.write();
56              final String newFile = FileUtils.readFileToString(temp1, "UTF-8");
57              assertTrue(newFile.contains("<false/>"));
58          } else if (cur1.contains("<false/>")) {
59              assertFalse(cur1.contains("<true/>"));
60              //Configurator.setStartAtLogin(temp, true);
61              settings.setRunAtSystemStart(true);
62              implementor.setStartAtLoginOsx(true);
63              modelIo.write();
64              final String newFile = FileUtils.readFileToString(temp1, "UTF-8");
65              assertTrue(newFile.contains("<true/>"));
66          } else {
67              fail("Model should have some start at login state");
68          }
69  
70          if (cur2.contains("X-GNOME-Autostart-enabled=true")) {
71              assertFalse(cur2.contains("<false/>"));
72              //Configurator.setStartAtLogin(temp, false);
73              settings.setRunAtSystemStart(false);
74              implementor.setStartAtLoginLinux(false);
75              modelIo.write();
76              final String newFile = FileUtils.readFileToString(temp2, "UTF-8");
77              assertTrue(newFile.contains("X-GNOME-Autostart-enabled=false"));
78          } else if (cur2.contains("X-GNOME-Autostart-enabled=false")) {
79              assertFalse(cur2.contains("X-GNOME-Autostart-enabled=true"));
80              //Configurator.setStartAtLogin(temp, true);
81              settings.setRunAtSystemStart(true);
82              implementor.setStartAtLoginLinux(true);
83              modelIo.write();
84              final String newFile = FileUtils.readFileToString(temp2, "UTF-8");
85              assertTrue(newFile.contains("X-GNOME-Autostart-enabled=true"));
86          } else {
87              fail("Gnome autostart should be in some state");
88          }
89      }
