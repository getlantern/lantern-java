39              {"name=value; domain=co.uk", "http://co.uk/"},
40              {"name=value; domain=.co.uk", "http://co.uk/"},
41              {"name=value", "http://127.0.0.1/"},

43          };

46              final HttpRequest req = new DefaultHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.GET, test[1]);

48              assertTrue(test[0] + ',' + test[1], policy.accepts(cookie));
49          }
50  
51          String acceptFalse[][] = new String[][]{
52              {"name=value; domain=abc.example.org;", "http://www.example.org"},
53              {"name=value; domain=abc.www.example.org;", "http://www.example.org"},
54              {"name=value; domain=ww.example.org;", "http://www.example.org/"},
55              {"name=value; domain=org;", "http://www.example.org"},
56              {"name=value; domain=.org;", "http://www.example.org"},
57              {"name=value; domain=facebook.com;", "http://www.example.org/"},
58              {"name=value; domain=.0.0.1;", "http://127.0.0.1/"},
59          };
60          for (String []test : acceptFalse) {
61              final Cookie cookie = createSetCookie(test[0], test[1]);
62              final HttpRequest req = new DefaultHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.GET, test[1]);
63              final CookieFilter policy = new CookieUtils.RFC6265SetCookieFilter(req);
64              assertFalse(test[0] + ',' + test[1], policy.accepts(cookie));

70       * tests rules for whether a cookie can/should be 
71       * sent to a given request uri based on domain.
72       */ 
73      @Test 
74      public void testCanBeSent() throws Exception {
75  
76          StoredCookie cookie;
77  
78          
79          /* 
80           * domain matching
81           */
82          
83          
84          cookie = new StoredCookie("name", "value");
85          cookie.setDomain("bar.example.org");
86          cookie.setPath("/");
87          cookie.setSecure(false);
88  
89          // when the host only flag is set, only the exact 
90          // domain name can have the cookie. 
91          cookie.setHostOnly(true);
92          // this exact match will pass
93          assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/")));
94          // all these non-exact matches will fail
95          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://foo.bar.example.org/")));
96          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://example.org/")));
97          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://www.example.org/")));
98          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://badbar.example.org/")));
99          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://elsewhere.com/")));
100         
101         // all else being equals, if host only is false, exact matching
102         // is not required, only 'domain matching'
103         cookie.setHostOnly(false);
104         // these domain match
105         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/")));
106         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://foo.bar.example.org/")));
107         // these should still fail.
108         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://example.org/")));
109         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://www.example.org/")));
110         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://badbar.example.org/")));
111         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://elsewhere.com/")));
112 
113         /*
114          * path matching 
115          */
116         cookie = new StoredCookie("name", "value");
117         cookie.setHostOnly(true);
118         cookie.setDomain("bar.example.org");
119         cookie.setSecure(false);
120         cookie.setPath("/foo");
121 
122         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo")));
123         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo/")));
124         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo/bar/baz")));
125         // these do not path match
126         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org")));
127         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/quux")));
128         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foobad")));
129         
130         // exact prefix is required, so this means something slightly different
131         cookie.setPath("/foo/");
132         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo/")));
133         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo/bar/baz")));
134         // these do not path match
135         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo")));
136         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org")));
137         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/quux")));
138         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foobad")));
139         
140         /*
141          * secure flag
142          */
143          cookie = new StoredCookie("name", "value");
144          cookie.setHostOnly(true);
145          cookie.setDomain("bar.example.org");
146          cookie.setPath("/");
147          cookie.setSecure(false);
148          
149          assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/")));
150     
151          // currently we always say false when the secure flag has 
152          // somehow been set and it is visible to us because we 
153          // are never dealing with the contents of a secure end-to-end
154          // communication (those are invisible to us)
155          cookie.setSecure(true); 
156          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/")));
157     }

335     /**
336      * tests construction of uris with non-ascii 
337      * hostnames.
338      */
339     @Test
340     public void testSafeURI() throws Exception {
341         URI uri;
342         
343         String uriString = "http://foö.bar.org/quux";
344         
345         // first, assert the problem exists, otherwise we 
346         // may not need this rigamarole. Hostname with 
347         // non-ascii characters will yield a null host.
348         uri = new URI("http://foö.bar.org/quux");
349         assertTrue(uri.getHost() == null);
350 
351         // using makeSafeURI we should get the A-Name version 
352         // of the hostname.
353         uri = CookieUtils.makeSafeURI(uriString);
354         assertTrue(uri.getHost().equals("xn--fo-gka.bar.org"));
355         assertTrue(uri.getPath().equals("/quux"));
356         
357         // additional non-ascii characters in the url 
358         // outside the hostname should be the same.
359         uriString = "http://foö.bar.org/qöux";
360         uri = CookieUtils.makeSafeURI(uriString);
361         assertTrue(uri.getHost().equals("xn--fo-gka.bar.org"));
362         assertTrue(uri.getPath().equals("/qöux"));
363         
364         // preserves other junks, just modifies host
365         uriString = "https://flöke:flöp@foö.bar.org:9191/qöux?zbfö=zboe&foo#glöp";
366         uri = CookieUtils.makeSafeURI(uriString);
367         assertTrue(uri.getHost().equals("xn--fo-gka.bar.org"));
368         assertTrue(uri.getPath().equals("/qöux"));
369         assertTrue(uri.getUserInfo().equals("flöke:flöp"));
370         assertTrue(uri.getScheme().equals("https"));
371         assertTrue(uri.getPort() == 9191);
372         assertTrue(uri.getQuery().equals("zbfö=zboe&foo"));
373         assertTrue(uri.getFragment().equals("glöp"));
374         assertTrue(uri.toString().equals("https://flöke:flöp@xn--fo-gka.bar.org:9191/qöux?zbfö=zboe&foo#glöp"));
375 
376         // should be idempotent
377         uri = CookieUtils.makeSafeURI(uri.toString());
378         assertTrue(uri.getHost().equals("xn--fo-gka.bar.org"));
379         assertTrue(uri.getPath().equals("/qöux"));
380         assertTrue(uri.getUserInfo().equals("flöke:flöp"));
381         assertTrue(uri.getScheme().equals("https"));
382         assertTrue(uri.getPort() == 9191);
383         assertTrue(uri.getQuery().equals("zbfö=zboe&foo"));
384         assertTrue(uri.getFragment().equals("glöp"));
385         assertTrue(uri.toString().equals("https://flöke:flöp@xn--fo-gka.bar.org:9191/qöux?zbfö=zboe&foo#glöp"));
